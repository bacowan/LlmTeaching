class Solution:
    def nFibonacci(self, N):
        # Initialize the first two terms of the Fibonacci sequence
        fib_series = [0, 1]
        
        # Add terms to the Fibonacci sequence until the last term is equal to or greater than N
        while fib_series[-1] < N:
            fib_series.append(fib_series[-1] + fib_series[-2])
            
        # Check if the last term is greater than N, if so, remove it
        if fib_series[-1] >= N:
            fib_series.pop()
        
        return fib_series
